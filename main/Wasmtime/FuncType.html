pub fn params(&self) -> Result<RArray, Error> {
    let len = self.inner.params().len();
    let mut params = self.inner.params();
    params.try_fold(RArray::with_capacity(len), |array, p| {
        array.push(p.to_sym()?)?;
        Ok(array)
    })
}
