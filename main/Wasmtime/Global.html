pub fn set(&self, value: Value) -> Result<(), Error> {
    self.inner
        .set(
            self.store.context_mut()?,
            value.to_wasm_val(self.value_type()?)?,
        )
        .map_err(|e| error!("{}", e))
        .and_then(|result| {
            self.retain_non_nil_extern_ref(value)?;
            Ok(result)
        })
}
