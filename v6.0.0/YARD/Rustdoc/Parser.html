def parse
  # - Find structs w/ @yard-rustdoc
  # - Find impls in these the struct
  #   - For methods with @yard-rustdoc
  #     - Parse methods
  #       - default to class vs instance method based on `self` first arg
  #       - Q: how to extract the signature?
  @namespaces = []
  @rustdoc_json["index"].filter_map do |_key, entry|
    next unless entry["crate_id"].zero?
    next unless entry["docs"]&.include?("@yard-rustdoc")
    next unless entry["kind"] == "struct"

  end
end
